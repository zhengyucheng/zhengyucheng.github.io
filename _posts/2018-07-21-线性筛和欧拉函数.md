### 程序

    #include <bits/stdc++.h>
    #define sc(p) scanf("%d",&p)
    const int N=10000100;
    using namespace std;
    int phi[N],prime[N]; bool vis[N];
    void init (int gd)
    {
        phi[1]=1;
        for (int i=2; i<=gd; i++)
        {
            if (!vis[i]) prime[++prime[0]]=i,phi[i]=i-1;
            for (int j=1; j<=prime[0]; j++)
            {
                long long k=prime[j]*i;
                if (k>gd) break;
                vis[k]=1;
                if (i%prime[j]==0)
                {
                    phi[k]=phi[i]*prime[j];
                    break;
                }
                else phi[k]=phi[i]*phi[prime[j]];
            }
        }

    }
    int main()
    {
        int n; sc(n); init (n);
        return 0;
    }

        int tot=0,root=0;
        void insert (int now,string sb)
        {
            for (int i=0; i<sb.length(); i++)
            {
                if (!t[now].nx[sb[i]-'a'])
                {
                    t[now].nx[sb[i]-'a']=++tot;
                    t[t[now].nx[sb[i]-'a']].fa=now;
                }
                now=t[now].nx[sb[i]-'a'];
            }
            t[now].cnt++;
        }
        int search (int now,string sb)
        {
            for (int i=0; i<sb.length(); i++)
            {
                now=t[now].nx[sb[i]-'a'];
                if (!now) return 0;
            }
            return t[now].cnt;
        }
        queue<int> q;
        void init (int now)
        {
            for (int i=0; i<26; i++)
            if (t[root].nx[i])
            {
                t[t[root].nx[i]].fail=root;
                q.push (t[root].nx[i]);
            }
            while (!q.empty())
            {
                int cur=q.front();
                q.pop();
                for (int i=0; i<26; i++)
                {
                    int nx=t[cur].nx[i];
                    if (nx)
                    {
                        t[nx].fail=t[t[cur].fail].nx[i];
                        q.push (nx);
                    }
                    else
                    t[cur].nx[i]=t[t[cur].fail].nx[i];
                }
            }
        }
        int query (string sb)
        {
            int l=sb.length ();
            int now=root,ans=0;
            for (int i=0; i<l; i++)
            {
                now=t[now].nx[sb[i]-'a'];
                for (int g=now; g&&t[g].cnt!=-1; g=t[g].fail)
                ans+=t[g].cnt,t[g].cnt=-1;
            }
            return ans;
        }

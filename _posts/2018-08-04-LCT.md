    void push (int x)
    {
        if (!x) return ;
        if (lazy(x))
        {
            swap (son(x)[0],son(x)[1]);
            if (son(x)[0]) lazy(son(x)[0])^=1;
            if (son(x)[1]) lazy(son(x)[1])^=1;
            lazy(x)=0;
        }
        if (addd(x))
        {
            w(x)+=addd(x),ma(x)+=addd(x);
            if (son(x)[0]) addd(son(x)[0])+=addd(x);
            if (son(x)[1]) addd(son(x)[1])+=addd(x);
            addd(x)=0;
        }
    }
    void update (int x)
    {
        if (!x) return ;
        ma(x)=w(x);
        int ls=son(x)[0],rs=son(x)[1];
        push (son(x)[0]);
        if (son(x)[0]) ma(x)=max (ma(x),ma(ls));
        push (son(x)[1]);
        if (son(x)[1]) ma(x)=max (ma(x),ma(rs));
    }
    int Getp (int x) { return son(fa(x))[1]==x; }
    void Rotate(int x)
    {
        int p=Getp(x), f=fa[x];
        if(fa[f]) son[fa[f]][Getp(f)] = x;
        fa[x] = fa[f];
        if(son[x][p^1]) fa[son[x][p^1]] = f;
        son[f][p] = son[x][p^1];
        son[x][p^1] = f, fa[f] = x;
        pathp[x] = pathp[f];
        pathp[f] = 0;
        update(f);
        update(x);
    }
    int stark[N];
    void splay(int x)
    {
        int to=0, y=x;
        stark[++to]=y;
        while(fa[y]) stark[++to]=fa[y], y=fa[y];
        while(to) push(stark[to--]);
        while(fa[x])
        {
            if(fa[fa[x]] && Getp(fa[x])==Getp(x)) Rotate(fa[x]);
            Rotate(x);
        }
        update(x);
    }
    void access (int x)
    {
        int y=0;
        while (x)
        {
            splay (x);
            if (son(x)[1])
            {
                fa(son(x)[1])=0;
                pathp(son(x)[1])=x;
            }
            son(x)[1]=y;
            if (y) fa(y)=x;
            pathp(y)=0;
            update (x);
            y=x;
            x=pathp(x);
        }
    }
    int getroot (int x)
    {
        access (x);
        splay (x);
        while (son(x)[0])
        {
            x=son(x)[0];
            push(x);
        }
        return x;
    }
    void evert (int x)
    {
        access (x);
        splay (x);
        lazy(x)^=1;
    }
    void link (int x,int y)
    {
        evert (y);
        splay (y);
        pathp(y)=x;

    void cut (int x,int y)
    {
        evert (x);
        access (y);
        splay (y);
        if (son(y)[0]) fa(son(y)[0])=0,son(y)[0]=0;
        update (y);
    }
    void adddd (int x,int y,int z)
    {
        evert (x);
        access (y);
        splay (y);
        addd(y)+=z;
    }
    int query (int x,int y)
    {
        evert (x);
        access (y);
        splay (y);

### 操作
#### access(x)
把节点`x`到树根链接到同一条树链上，再让`x`成为链的底端。

    void access (int x)
    {
        int y=0;
        while (x)
        {
            splay (x);
            if (son(x)[1])
            {
                fa(son(x)[1])=0;
                pathp(son(x)[1])=x;
            }
            son(x)[1]=y;
            if (y) fa(y)=x;
            pathp(y)=0;
            update (x);
            y=x;
            x=pathp(x);
        }
    }
    
#### getroot(x)
找`x`所在树的根。（原树，不是`splay`树，别忘了是森林。）

    int getroot (int x)
    {
        access (x);
        splay (x);
        while (son(x)[0])
        {
            x=son(x)[0];
            push(x);
        }
        return x;
    }
    
#### evert(x)
让`x`成为所在原树的树根。

    void evert (int x)
    {
        access (x);
        splay (x);
        lazy(x)^=1;
    }

#### link(u,v)
连接`u`和`v`。

    void link (int x,int y)
    {
        evert (y);
        splay (y);
        pathp(y)=x;
    }

#### cut(u,v)
断开u和v之间的边。首先要保证有边。

    void cut (int x,int y)
    {
        evert (x);
        access (y);
        splay (y);
        if (son(y)[0]) fa(son(y)[0])=0,son(y)[0]=0;
        update (y);
    }
    
#### 其他
只要把一个点`evert`，另一个`access`，此时树上路径就就是一条树链了，而且还在一棵`splay`树中。  
将根的标记取出/打标记即可。

### 关于翻转标记
要先把`x`到根依次放入栈中，然后一个一个取出来把翻转标记处理掉。  
不会增加时间复杂度。
### splay上标记维护
需要下推的标记(如翻转)，在旋转之前就要下推，同时将它的左右儿数据子的也先下推。  
需要统计的(如求和)，每次旋转只动`2`个点，旋转完之后，先维护较深的点（原来的父亲），再维护浅的点（转动的点）。  
其他问题和线段树一样。
